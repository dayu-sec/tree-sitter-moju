==================
Dataflow, target, binding
==================

dataflow OrderDataflow {
  node<message> SubmitOrder
  node<storage> OrderStore
  edge<write> SubmitOrder -> OrderStore {
    data Order.id, Order.status
  }
}

target<bin, http> {
  name "checkout_service"
}

target<mod> {
  path "src/checkout"
}

bind CheckoutInterface by protocol<http> {
  entry checkout {
    route POST "/checkout/orders"
    input SubmitOrder
    output CheckoutAccepted status 202
  }
}

bind OrderStore to sqldb<postgres> {
  table orders
  primary_key id
  index user
}

bind CheckoutConfig to file<toml> {
  path {
    env CONFIG_PATH
    default "/etc/config.toml"
  }
}
