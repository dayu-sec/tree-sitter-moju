struct Order {
  unique id
  status: OrderStatus
}

state OrderStatus {
  Created
  Paid
}

event PaymentSucceeded {
  order
}

lifecycle OrderLifecycle for Order.status {
  initial Created
  transition Created -> Paid on PaymentSucceeded
}
