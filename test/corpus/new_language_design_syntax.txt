struct Order implements Validatable, Activable {
  unique id: String
  amount: Int
  status: OrderStatus

  op<sync> InventoryReserved(cart, items);
  op InventoryReleased(cart, items);

  send OrderCreated -> ConfigHistoryInfo {
    orderId = this.id
    amount = this.amount
  }
}

scenario PlaceOrder {
  participant Order
  participant PaymentGateway

  Order send OrderCreated -> ConfigHistoryInfo

  alt Paid when equals(Order.status, Paid) {
    PaymentGateway send<async> PaymentSucceeded -> Order
  }
}

retry_policy ClusterSyncRetry {
  max 6
  backoff exponential from 500ms factor 2
}

flow ConfigPublish when not_empty(PublishConfig.content) {
  trigger PublishConfig

  step Persist -> ConfigDataChangeEvent {
    call ConfigPersist.PublishConfig {
      retry ClusterSyncRetry
      timeout 30s -> ConfigPublishFailed
    }

    match {
      not_empty(PublishConfig.betaIps) -> ConfigBetaPublish
      _ -> FormalPublish
    }
  }

  react<async> ConfigDataChangeEvent {
    step ClusterSync {
      call ConfigNotify.NotifyClusterMembers with {
        retry {
          max 3
          backoff linear from 1s step 2s
        }
      }
    }
  }
}

verify Success for flow ConfigPublish {
  given PublishConfig.content = "x"
  when PublishConfig
  expect PublishConfig.content == "x"
}
